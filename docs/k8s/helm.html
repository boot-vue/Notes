# 自定义 chart

  1. Chart.yaml
# name version必须有
name: xxxx
version: 1.0.0
  1. 创建 k8s yaml 文件

./templates/deployment.yaml

apiVersion: apps/v1
kind: Deployment
#  ...

./templates/service.yaml

# service yaml配置
  1. 启动
helm create xxx  # 创建chart目录和chart用到的公共文件目录结构

helm lint xxx   # 检查配置

helm install --set name=xxx .

helm install -n xxx -namespace vvv .

helm list

helm upgrade xxxx
